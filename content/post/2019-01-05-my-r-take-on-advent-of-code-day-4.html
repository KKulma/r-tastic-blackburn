clean_input <- raw_input %>% strsplit(‘’) %>% unlist() %>% as_tibble() %>% mutate(timestamp = ymd_hm(str_extract(value, ‘[:digit:]+-[:digit:]+-[:digit:]+..[:digit:]+:[:digit:]+’)), action = str_extract(value, ‘[:alpha:]+..[:alpha:]+’), guard_num = ifelse(str_detect(value, ‘#’), str_extract(value, ‘#[:digit:]+’), NA), date = ymd(str_sub(timestamp, start = 1, end = 10)), minute = minute(timestamp) ) %>% arrange(timestamp) %>% fill(guard_num) # fill in missing guard numbers

who sleeps most

clean_input %>% filter(action != ‘Guard’) %>% group_by(guard_num, date) %>% mutate(time_asleep = ifelse(action == ‘wakes up’, minute - lag(minute), NA ) #calculate time asleep ) %>% group_by(guard_num) %>% na.omit() %>% summarise(total_asleep = sum(time_asleep)) %>% # sum it arrange(desc(total_asleep)) %>% head(1)

guard_data <- clean_input %>% filter(action != ‘Guard’) %>% filter(guard_num == ‘#409’) %>% # pick the winner arrange(timestamp) %>% spread(action, minute) %>% # prep the data for sequences rename(falls_asleep = falls asleep, wakes_up = wakes up) %>% mutate(falls_asleep = ifelse(!is.na(wakes_up), lag(falls_asleep), falls_asleep )) %>% na.omit()

function that generates sequence of minutes between falling asslep and waking up

time_seq <- function(falls, wakes) { result <- seq(falls, wakes) return(result) }
